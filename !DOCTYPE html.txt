  <script>
    // タブ切り替え
    const calendarTab = document.getElementById("calendar-tab");
    const taskTab = document.getElementById("task-tab");
    const calendar = document.getElementById("calendar");
    const taskView = document.getElementById("task-view");

    calendarTab.addEventListener("click", () => {
      calendarTab.classList.add("active");
      taskTab.classList.remove("active");
      calendar.style.display = "block";
      taskView.style.display = "none";
    });

    taskTab.addEventListener("click", () => {
      taskTab.classList.add("active");
      calendarTab.classList.remove("active");
      taskView.style.display = "block";
      calendar.style.display = "none";
    });

    // カレンダー作成
    const monthYear = document.getElementById("monthYear");
    const calendarGrid = document.getElementById("calendarGrid");
    let currentDate = new Date();

    function renderCalendar(date) {
      const year = date.getFullYear();
      const month = date.getMonth();
      monthYear.textContent = `${year}年 ${month + 1}月`;

      calendarGrid.innerHTML = "";
      const firstDay = new Date(year, month, 1).getDay();
      const lastDate = new Date(year, month + 1, 0).getDate();

      for (let i = 0; i < firstDay; i++) {
        calendarGrid.innerHTML += `<div></div>`;
      }

      for (let d = 1; d <= lastDate; d++) {
        const div = document.createElement("div");
        div.className = "day";
        div.textContent = d;
        div.addEventListener("click", () => {
          alert(`${month + 1}月${d}日の課題を確認してください`);
        });
        calendarGrid.appendChild(div);
      }
    }

    renderCalendar(currentDate);

    document.getElementById("prevMonth").addEventListener("click", () => {
      currentDate.setMonth(currentDate.getMonth() - 1);
      renderCalendar(currentDate);
    });

    document.getElementById("nextMonth").addEventListener("click", () => {
      currentDate.setMonth(currentDate.getMonth() + 1);
      renderCalendar(currentDate);
    });

    // 終了タスク表示切替
    const toggleCompleted = document.getElementById("toggleCompleted");
    const completedTable = document.getElementById("completed-table");
    toggleCompleted.addEventListener("change", () => {
      completedTable.style.display = toggleCompleted.checked ? "table" : "none";
    });

    // 完了→戻す機能
    const restoreCells = document.querySelectorAll(".restore");
    const taskTable = document.getElementById("task-table");

    restoreCells.forEach(cell => {
      cell.addEventListener("click", (e) => {
        const row = e.target.closest("tr");
        const newRow = row.cloneNode(true);
        newRow.querySelector(".restore")?.remove(); // 「戻す」文言削除
        taskTable.appendChild(newRow);
        row.remove();
      });
    });
  </script>
